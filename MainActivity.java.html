package com.example.bharat;

import androidx.appcompat.app.AppCompatActivity;

import android.os.Bundle;
import android.util.Log;
import android.view.View;
import android.widget.Button;
import android.widget.EditText;
import android.widget.TextView;

public class MainActivity extends AppCompatActivity {

    @Override
    protected void onCreate(Bundle savedInstanceState){
        EditText num1, num2;
        Button sum;
        TextView res;
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);
        Log.d("MainActivity", "onCreateInstance");
        setContentView(R.layout.activity_main);
        num1=findViewById(R.id.ed1);
        num2=findViewById(R.id.ed2);
        sum=findViewById(R.id.add);
        res=findViewById(R.id.tv1);


        sum.setOnClickListener(new View.OnClickListener() {
            @Override
            public void onClick(View view) {
                int n1 = Integer.parseInt(num1.getText().toString());
                int n2 = Integer.parseInt(num2.getText().toString());
                int s = n1+n2;
                res.setText("Result is:"+s);
            }
        });
    }

    @Override
    protected void onStart() {
        super.onStart();
        Log.d("MainActivity", "onStartInstance");
    }

    @Override
    protected void onResume() {
        super.onResume();
        Log.d("MainActivity", "onResumeInstance");
    }

    @Override
    protected void onPause() {
        super.onPause();
        Log.d("MainActivity", "onPauseInstance");
    }

    @Override
    protected void onDestroy() {
        super.onDestroy();
        Log.d("MainActivity", "onDestroyInstance");
    }

    @Override
    protected void onRestart() {
        super.onRestart();
        Log.d("MainActivity", "onRestartInstance");
    }
